# 查找重复记录
select o.hz, oc.dupeCount, o.id
from zwzd o
inner join (
    SELECT hz, COUNT(*) AS dupeCount
    FROM zwzd
    GROUP BY hz
    HAVING COUNT(*) > 1
) oc on o.hz = oc.hz
